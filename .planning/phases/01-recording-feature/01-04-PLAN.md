/**
 * Current language store.
 * Writable store that can be subscribed to for reactive language changes.
 */
export const currentLanguage = writable<Language>('en');

/**
 * Initialize language on app startup.
 * Called in +layout.svelte onMount.
 * 1. Check localStorage for saved language preference
 * 2. If not found, detect from browser locale
 * 3. If locale not recognized, default to English
 * @returns The initialized language
 */
export function initLanguage(): Language {
  let language: Language = 'en';

  // Step 1: Check localStorage for saved preference
  try {
    const saved = localStorage.getItem('nhonly-language');
    if (saved === 'vi' || saved === 'en') {
      language = saved;
    }
  } catch (error) {
    // localStorage not available (private browsing, etc.)
    console.debug('localStorage not available, using browser locale detection');
  }

  // Step 2: If not saved, detect from browser locale
  if (!localStorage.getItem('nhonly-language')) {
    const browserLocale = navigator.language?.toLowerCase() ?? 'en';
    if (browserLocale.startsWith('vi')) {
      language = 'vi';
    }
  }

  // Set the store
  currentLanguage.set(language);
  return language;
}

/**
 * Set language and persist to localStorage.
 * Called when user clicks language toggle.
 * @param lang - Language to set ('vi' or 'en')
 */
export function setLanguage(lang: Language): void {
  if (lang !== 'vi' && lang !== 'en') {
    console.warn(`Invalid language: ${lang}, using English`);
    lang = 'en';
  }

  currentLanguage.set(lang);

  // Persist to localStorage
  try {
    localStorage.setItem('nhonly-language', lang);
  } catch (error) {
    console.debug('Failed to persist language to localStorage:', error);
  }
}

/**
 * Get translation for a key in the current language.
 * Called by UI components to get translated strings.
 * Success: key exists in current language, returns string
 * Failure: returns fallback to English if key missing, then returns key itself as last resort
 * @param key - Translation key (e.g., 'tell_story')
 * @returns Translated string, or key itself if missing
 */
export function t(key: string): string {
  const lang = get(currentLanguage);

  // Get string in current language
  const translated = strings[lang]?.[key];
  if (translated) {
    return translated;
  }

  // Fallback to English
  const english = strings['en']?.[key];
  if (english) {
    console.warn(`Missing translation for key "${key}" in ${lang}, using English`);
    return english;
  }

  // Last resort: return key itself (helps identify missing translations)
  console.warn(`Missing translation for key "${key}" in both ${lang} and English`);
  return key;
}

/**
 * Map error types to localized messages.
 * Called by +page.svelte when catching errors.
 * @param error - Caught error
 * @param language - Current language ('vi' or 'en')
 * @returns User-friendly localized error message
 */
export function getLocalizedErrorMessage(error: any, language: Language = get(currentLanguage)): string {
  const errorMessage = error?.message ?? String(error);

  if (errorMessage.includes('Permission') || errorMessage.includes('NotAllowedError')) {
    return language === 'vi'
      ? strings['vi'].microphone_permission_denied
      : strings['en'].microphone_permission_denied;
  }

  if (errorMessage.includes('quota') || errorMessage.includes('full')) {
    return language === 'vi'
      ? strings['vi'].storage_full
      : strings['en'].storage_full;
  }

  if (errorMessage.includes('not supported')) {
    return language === 'vi'
      ? strings['vi'].browser_old
      : strings['en'].browser_old;
  }

  return language === 'vi'
    ? strings['vi'].recording_failed
    : strings['en'].recording_failed;
}
