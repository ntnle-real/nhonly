// MARK: SYSTEM(Audio) -> Web Audio API Waveform Analysis
// Purpose: Extract frequency data from active recording for real-time visualization
// Success: AnalyserNode captures frequency data, returns Uint8Array at 30+ FPS
// Failure: Web Audio API unavailable or recording stream not connected

import { ObservationSession, createObservationSession } from './obs';
import { BrowserError } from './errors';

/**
 * Waveform analyzer service.
 * Manages AnalyserNode connection to recording stream and frequency data extraction.
 */
interface WaveformAnalyzer {
  audioContext: AudioContext | null;
  analyser: AnalyserNode | null;
  animationFrameId: number | null;
  dataArray: Uint8Array | null;
  isAnalyzing: boolean;
}

const analyzer: WaveformAnalyzer = {
  audioContext: null,
  analyser: null,
  animationFrameId: null,
  dataArray: null,
  isAnalyzing: false
};

/**
 * Create or get AudioContext.
 * AudioContext is expensive to create, so we reuse it across the app lifetime.
 */
function getAudioContext(): AudioContext {
  if (analyzer.audioContext) {
    return analyzer.audioContext;
  }

  const audioContextClass = (window as any).AudioContext || (window as any).webkitAudioContext;
  if (!audioContextClass) {
    throw new BrowserError('Web Audio API not supported in this browser', ['Web Audio API']);
  }

  analyzer.audioContext = new audioContextClass();
  return analyzer.audioContext;
}

/**
 * Start waveform analysis on a MediaStream.
 * Called when recording starts. Connects AnalyserNode to the stream source.
 * @param stream - MediaStream from getUserMedia (passed from recording.ts)
 * @throws BrowserError if Web Audio API not available
 */
export function startWaveformAnalysis(
  stream: MediaStream,
  obs: ObservationSession = createObservationSession()
): void {
  obs.read('waveform_stream', { streamTracks: stream.getTracks().length });
  obs.step('create_audio_context');

  try {
    const audioContext = getAudioContext();
    obs.observe('audio_context_ready', `context state=${audioContext.state}`);

    obs.step('create_analyser_node');
    analyzer.analyser = audioContext.createAnalyser();
    analyzer.analyser.fftSize = 256; // 2^8 frequency bins (128 values)
    analyzer.analyser.smoothingTimeConstant = 0.8; // Smooth transitions
    analyzer.dataArray = new Uint8Array(analyzer.analyser.frequencyBinCount);

    obs.observe('analyser_created', `fftSize=${analyzer.analyser.fftSize}, bins=${analyzer.analyser.frequencyBinCount}`);

    obs.step('connect_stream_to_analyser');
    const source = audioContext.createMediaStreamSource(stream);
    source.connect(analyzer.analyser);

    obs.observe('stream_connected', 'MediaStreamAudioSourceNode connected to analyser');

    analyzer.isAnalyzing = true;
    obs.return_('waveform_analysis_started', undefined);
  } catch (error) {
    obs.observe('analyser_error', (error as Error).message);
    throw error;
  }
}

/**
 * Stop waveform analysis.
 * Called when recording stops or UI unmounts.
 * Cancels animation frame and resets analyzer state.
 */
export function stopWaveformAnalysis(obs: ObservationSession = createObservationSession()): void {
  obs.step('stop_waveform_analysis');

  if (analyzer.animationFrameId !== null) {
    cancelAnimationFrame(analyzer.animationFrameId);
    analyzer.animationFrameId = null;
  }

  analyzer.isAnalyzing = false;
  analyzer.analyser = null;
  analyzer.dataArray = null;

  obs.observe('waveform_analysis_stopped', 'animation frame cancelled, analyser reset');
  return obs.return_('analysis_stopped', undefined);
}

/**
 * Get current frequency data from analyser.
 * Called by waveform.svelte component on each animation frame.
 * Returns Uint8Array where each byte is a frequency bin (0-255).
 * Lower frequencies (bass) are at indices 0-32, higher (treble) at 128+.
 * @returns Copy of current frequency data, or null if not analyzing
 */
export function getFrequencyData(): Uint8Array | null {
  if (!analyzer.analyser || !analyzer.dataArray) {
    return null;
  }

  analyzer.analyser.getByteFrequencyData(analyzer.dataArray);
  // Return a copy so UI doesn't hold mutable reference
  return new Uint8Array(analyzer.dataArray);
}

/**
 * Check if waveform analysis is active.
 */
export function isAnalyzing(): boolean {
  return analyzer.isAnalyzing;
}

/**
 * Cleanup and resume context if paused.
 * AudioContext may pause in background, so resume on navigation back.
 */
export function ensureAudioContextRunning(): void {
  if (analyzer.audioContext && analyzer.audioContext.state === 'suspended') {
    analyzer.audioContext.resume().catch((error) => {
      console.warn('Failed to resume AudioContext:', error);
    });
  }
}
```

- `getFrequencyData()` returns Uint8Array updated by AnalyserNode
- `stopWaveformAnalysis()` cleans up and cancels animation frame
- `ensureAudioContextRunning()` resumes paused AudioContext if needed

<script lang="ts">
  import { getFrequencyData, startWaveformAnalysis, stopWaveformAnalysis, ensureAudioContextRunning } from '$lib/waveform';
  import { onMount, onDestroy } from 'svelte';

  // MARK: props
  let {
    stream = $bindable<MediaStream | null>(null),
    isRecording = $bindable(false)
  }: {
    stream: MediaStream | null;
    isRecording: boolean;
  } = $props();

  let canvas: HTMLCanvasElement;
  let animationFrameId: number | null = null;

  // MARK: lifecycle
  onMount(() => {
    ensureAudioContextRunning();
  });

  onDestroy(() => {
    if (animationFrameId !== null) {
      cancelAnimationFrame(animationFrameId);
    }
    stopWaveformAnalysis();
  });

  // MARK: reactive: when stream available, start analyzing
  $effect(() => {
    if (stream && isRecording) {
      try {
        startWaveformAnalysis(stream);
        drawWaveform();
      } catch (error) {
        console.error('Failed to start waveform analysis:', error);
      }
    } else {
      stopWaveformAnalysis();
    }
  });

  /**
   * Draw waveform using frequency data from AnalyserNode.
   * Called via requestAnimationFrame at 30+ FPS.
   * Draws bars for each frequency bin (low=left/green, high=right/red).
   */
  function drawWaveform() {
    if (!canvas || !isRecording) {
      return;
    }

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    // Get frequency data (0-255 for each bin)
    const dataArray = getFrequencyData();
    if (!dataArray) {
      // Still analyzing but no data yet, schedule next frame
      animationFrameId = requestAnimationFrame(drawWaveform);
      return;
    }

    // Clear canvas
    const width = canvas.width;
    const height = canvas.height;
    ctx.fillStyle = '#f0f0f0';
    ctx.fillRect(0, 0, width, height);

    // Draw frequency bars
    // Lower frequencies = left side (green), higher = right side (red)
    const barWidth = width / dataArray.length;
    const centerY = height / 2;

    dataArray.forEach((value, index) => {
      // Normalize frequency value (0-255) to bar height
      const barHeight = (value / 255) * (height * 0.4);

      // Color gradient: green (low) → yellow (mid) → red (high)
      const hue = (index / dataArray.length) * 120; // 0=red, 120=green
      ctx.fillStyle = `hsl(${hue}, 100%, 50%)`;

      // Draw bar (top and bottom mirrored around center)
      ctx.fillRect(index * barWidth, centerY - barHeight, barWidth - 1, barHeight);
      ctx.fillRect(index * barWidth, centerY, barWidth - 1, barHeight);
    });

    animationFrameId = requestAnimationFrame(drawWaveform);
  }

  // MARK: rendering
  // Canvas size responsive to viewport
  let containerWidth = 100; // percentages aren't allowed in canvas
  let containerHeight = 150;
</script>

<div class="waveform-container">
  <canvas
    bind:this={canvas}
    width={Math.min(window.innerWidth - 40, 600)}
    height={containerHeight}
    class="waveform-canvas"
    aria-label="Audio waveform visualization"
  />
</div>

// MARK: SYSTEM(Audio) -> Audio Preview Blob URL Management
// Purpose: Create and revoke blob URLs for audio preview playback without memory leaks
// Success: blob URL created from audio Blob, playable in <audio> element
// Failure: blob URL revoked on component destroy, preventing memory leaks

import { ObservationSession, createObservationSession } from './obs';

/**
 * Track created blob URLs for cleanup.
 * Maps blob to its URL for later revocation.
 */
const blobURLs = new Map<Blob, string>();

/**
 * Create a blob URL from audio Blob for preview playback.
 * Called when recording stops and user sees preview option.
 * Must be revoked with revokePreviewURL() to prevent memory leaks.
 * @param audioBlob - Audio Blob from recording
 * @returns Object URL that can be used in <audio src="">
 */
export function createPreviewURL(
  audioBlob: Blob,
  obs: ObservationSession = createObservationSession()
): string {
  obs.read('audio_blob', { size: audioBlob.size, type: audioBlob.type });
  obs.step('create_blob_url');

  try {
    const url = URL.createObjectURL(audioBlob);
    blobURLs.set(audioBlob, url);
    obs.observe('blob_url_created', url);
    return obs.return_('preview_url', url);
  } catch (error) {
    obs.observe('blob_url_error', (error as Error).message);
    throw new Error(`Failed to create preview URL: ${(error as Error).message}`);
  }
}

/**
 * Revoke a blob URL to release memory.
 * Called when preview is closed or component unmounts.
 * Critical for preventing memory leaks with large audio files.
 * @param audioBlob - Original Blob (to look up URL)
 */
export function revokePreviewURL(
  audioBlob: Blob,
  obs: ObservationSession = createObservationSession()
): void {
  obs.read('audio_blob_to_revoke', { size: audioBlob.size });
  obs.step('revoke_blob_url');

  const url = blobURLs.get(audioBlob);
  if (url) {
    URL.revokeObjectURL(url);
    blobURLs.delete(audioBlob);
    obs.observe('blob_url_revoked', url);
  } else {
    obs.observe('url_not_found', 'blob URL not tracked');
  }

  return obs.return_('revocation_complete', undefined);
}

/**
 * Revoke all tracked blob URLs.
 * Called on app shutdown or cleanup.
 */
export function revokeAllPreviewURLs(
  obs: ObservationSession = createObservationSession()
): void {
  obs.step('revoke_all_blob_urls');

  for (const [blob, url] of blobURLs.entries()) {
    URL.revokeObjectURL(url);
  }

  const count = blobURLs.size;
  blobURLs.clear();

  obs.observe('all_urls_revoked', `revoked ${count} URLs`);
  return obs.return_('all_revoked', undefined);
}
```

Key features:
- `createPreviewURL()` creates object URL from Blob
- `revokePreviewURL()` revokes URL to free memory
- `revokeAllPreviewURLs()` cleanup helper
- Map tracking prevents duplicate URLs

<script lang="ts">
  import { t, setLanguage, currentLanguage } from '$lib/i18n';
  import {
    startRecording,
    pauseRecording,
    resumeRecording,
    stopRecording,
    getElapsedMs,
    isRecording,
    isPausedRecording,
    cancelRecording
  } from '$lib/recording';
  import { initDatabase, saveStory } from '$lib/archive';
  import { createPreviewURL, revokePreviewURL } from '$lib/preview';
  import { checkBrowserCapabilities } from '$lib/capabilities';
  import { PermissionError, BrowserError } from '$lib/errors';
  import Waveform from '$lib/waveform.svelte';
  import { onMount, onDestroy } from 'svelte';

  // MARK: state
  type RecordingState = 'idle' | 'recording' | 'paused' | 'stopped' | 'saving' | 'saved' | 'error';

  let recordingState: RecordingState = $state('idle');
  let elapsedMs = $state(0);
  let audioBlob: Blob | null = $state(null);
  let audioPreviewURL: string | null = $state(null);
  let storyTitle = $state('');
  let statusMessage = $state('');
  let errorMessage = $state('');
  let recordingStream: MediaStream | null = $state(null);
  let browserCapabilities = $state.snapshot({ isSupported: false, browser: 'unknown', missingAPIs: [] });

  // MARK: initialization
  onMount(async () => {
    // Check browser capabilities on app start
    browserCapabilities = checkBrowserCapabilities();
    if (!browserCapabilities.isSupported) {
      recordingState = 'error';
      errorMessage = `Your browser (${browserCapabilities.browser}) is not supported. Please use Chrome, Firefox, Safari, or Edge.`;
      return;
    }

    // Initialize database
    try {
      await initDatabase();
    } catch (error) {
      console.error('Failed to initialize database:', error);
      errorMessage = 'Failed to initialize storage. Please check browser settings.';
    }

    // Update elapsed time during recording
    const interval = setInterval(() => {
      if (isRecording() || isPausedRecording()) {
        elapsedMs = getElapsedMs();
      }
    }, 100);

    return () => clearInterval(interval);
  });

  onDestroy(() => {
    // Cleanup: cancel recording if still active
    if (isRecording() || isPausedRecording()) {
      cancelRecording();
    }
    // Cleanup: revoke preview URL if exists
    if (audioBlob && audioPreviewURL) {
      revokePreviewURL(audioBlob);
    }
  });

  // MARK: handler — tell story button
  async function handleStartRecording() {
    try {
      recordingState = 'recording';
      errorMessage = '';
      statusMessage = '';
      await startRecording();
    } catch (error) {
      recordingState = 'error';
      if (error instanceof PermissionError) {
        errorMessage = error.message;
      } else {
        errorMessage = 'Failed to start recording. Please check your microphone.';
      }
    }
  }

  // MARK: handler — pause button
  function handlePause() {
    try {
      pauseRecording();
      recordingState = 'paused';
    } catch (error) {
      errorMessage = (error as Error).message || 'Failed to pause recording';
    }
  }

  // MARK: handler — resume button
  function handleResume() {
    try {
      resumeRecording();
      recordingState = 'recording';
    } catch (error) {
      errorMessage = (error as Error).message || 'Failed to resume recording';
    }
  }

  // MARK: handler — stop recording button
  async function handleStopRecording() {
    try {
      audioBlob = await stopRecording();
      recordingState = 'stopped';
      storyTitle = '';
      errorMessage = '';
      statusMessage = '';

      // Create preview URL for playback
      if (audioBlob) {
        audioPreviewURL = createPreviewURL(audioBlob);
      }
    } catch (error) {
      recordingState = 'error';
      errorMessage = 'Failed to stop recording. Try again.';
    }
  }

  // MARK: handler — archive button
  async function handleArchive() {
    if (!audioBlob || !storyTitle.trim()) {
      errorMessage = t('please_enter_title');
      return;
    }

    try {
      recordingState = 'saving';
      errorMessage = '';
      await saveStory(storyTitle, audioBlob);
      recordingState = 'saved';
      statusMessage = t('story_archived');
      storyTitle = '';

      // Cleanup preview URL
      if (audioPreviewURL) {
        revokePreviewURL(audioBlob);
        audioPreviewURL = null;
      }
      audioBlob = null;

      // Reset after 3 seconds
      setTimeout(() => {
        recordingState = 'idle';
        statusMessage = '';
      }, 3000);
    } catch (error) {
      recordingState = 'error';
      errorMessage = 'Failed to save story. Please check storage space.';
    }
  }

  // MARK: handler — language toggle
  function toggleLanguage() {
    const newLang = $currentLanguage === 'en' ? 'vi' : 'en';
    setLanguage(newLang);
  }

  // MARK: helper — format time
  function formatTime(ms: number): string {
    const seconds = Math.floor(ms / 1000);
    const minutes = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${minutes}:${secs.toString().padStart(2, '0')}`;
  }
</script>

    <!-- General error -->
    <section class="error-section">
      <h2>{t('error_occurred')}</h2>
      <p>{errorMessage}</p>
      <button onclick={() => (recordingState = 'idle', errorMessage = '')}>
        {t('try_again')}
      </button>
    </section>

    <!-- MARK: idle state — tell story button -->
  {:else if recordingState === 'idle'}
    <section class="idle-section">
      <button onclick={handleStartRecording} class="primary-button">
        {t('tell_story')}
      </button>
    </section>

    <!-- MARK: recording state — timer, waveform, pause/resume/stop -->
  {:else if recordingState === 'recording' || recordingState === 'paused'}
    <section class="recording-section">
      <!-- Recording indicator -->
      <div class="recording-indicator {recordingState === 'paused' ? 'paused' : ''}">
        <div class="indicator-dot"></div>
        <span>{recordingState === 'paused' ? t('paused') : t('recording')}</span>
      </div>

      <!-- Timer display -->
      <div class="timer-display">
        {formatTime(elapsedMs)}
      </div>

      <!-- Waveform visualization -->
      <Waveform {recordingStream} isRecording={recordingState === 'recording'} />

      <!-- Control buttons -->
      <div class="button-group">
        {#if recordingState === 'recording'}
          <button onclick={handlePause} class="secondary-button">
            {t('pause')}
          </button>
        {/if}

        {#if recordingState === 'paused'}
          <button onclick={handleResume} class="secondary-button">
            {t('resume')}
          </button>
        {/if}

        <button onclick={handleStopRecording} class="stop-button">
          {t('stop_recording')}
        </button>
      </div>

      <p class="status-text">{t('auto_saving')}</p>
    </section>

    <!-- MARK: stopped state — preview, title, archive -->
  {:else if recordingState === 'stopped'}
    <section class="stopped-section">
      <!-- Audio preview player -->
      {#if audioPreviewURL}
        <div class="preview-player">
          <h3>{t('preview')}</h3>
          <audio controls {src}={audioPreviewURL} class="audio-player" />
        </div>
      {/if}

      <!-- Title input -->
      <div class="form-group">
        <label for="story-title">{t('story_title')}</label>
        <input
          id="story-title"
          type="text"
          bind:value={storyTitle}
          placeholder={t('enter_title')}
          class="title-input"
        />
      </div>

      <!-- Archive button -->
      <button onclick={handleArchive} class="primary-button">
        {t('archive_button')}
      </button>

      <!-- Option to record again -->
      <button
        onclick={() => {
          recordingState = 'idle';
          audioBlob = null;
          if (audioPreviewURL) {
            revokePreviewURL(audioBlob!);
            audioPreviewURL = null;
          }
        }}
        class="secondary-button"
      >
        {t('record_again')}
      </button>
    </section>

    <!-- MARK: saving state — progress indicator -->
  {:else if recordingState === 'saving'}
    <section class="saving-section">
      <p>{t('saving_story')}...</p>
      <div class="spinner"></div>
    </section>

    <!-- MARK: saved state — confirmation -->
  {:else if recordingState === 'saved'}
    <section class="saved-section">
      <div class="success-indicator">
        <span class="checkmark">✓</span>
      </div>
      <h2>{t('story_archived')}</h2>
      <p>{t('save_confirmation')}</p>
    </section>
  {/if}

  <!-- Error/status messages -->
  {#if errorMessage}
    <div class="error-message" role="alert">
      {errorMessage}
    </div>
  {/if}

  {#if statusMessage}
    <div class="status-message" role="status">
      {statusMessage}
    </div>
  {/if}
</div>
